# 🛡️ Manejo de Errores

El manejo de errores adecuado es lo que separa al código amateur del código profesional.

---

## `try / catch / finally`

```javascript
try {
  // Código que puede fallar
  const datos = JSON.parse('{"nombre": "Samuel"}');
  console.log(datos.nombre);
} catch (error) {
  // Se ejecuta si hay un error
  console.error("Error al parsear:", error.message);
} finally {
  // Se ejecuta SIEMPRE (éxito o error)
  console.log("Proceso finalizado");
}
```

## El objeto Error

```javascript
const error = new Error("Algo salió mal");

console.log(error.message);  // "Algo salió mal"
console.log(error.name);     // "Error"
console.log(error.stack);    // Stack trace completo

// Tipos de error nativos
new TypeError("Tipo incorrecto");
new RangeError("Fuera de rango");
new ReferenceError("Variable no definida");
new SyntaxError("Sintaxis inválida");
new URIError("URI malformada");
```

## `throw` — Lanzar errores

```javascript
function dividir(a, b) {
  if (typeof a !== "number" || typeof b !== "number") {
    throw new TypeError("Los argumentos deben ser números");
  }
  if (b === 0) {
    throw new RangeError("No se puede dividir entre 0");
  }
  return a / b;
}

try {
  console.log(dividir(10, 0));
} catch (error) {
  if (error instanceof RangeError) {
    console.error("Error de rango:", error.message);
  } else if (error instanceof TypeError) {
    console.error("Error de tipo:", error.message);
  } else {
    throw error; // Re-lanzar errores inesperados
  }
}
```

## Custom Errors (Errores personalizados)

```javascript
class AppError extends Error {
  constructor(message, statusCode, code) {
    super(message);
    this.name = "AppError";
    this.statusCode = statusCode;
    this.code = code;
  }
}

class NotFoundError extends AppError {
  constructor(recurso) {
    super(`${recurso} no encontrado`, 404, "NOT_FOUND");
    this.name = "NotFoundError";
  }
}

class ValidationError extends AppError {
  constructor(campo, detalle) {
    super(`Validación fallida en "${campo}": ${detalle}`, 400, "VALIDATION_ERROR");
    this.name = "ValidationError";
    this.campo = campo;
  }
}

class AuthError extends AppError {
  constructor(message = "No autorizado") {
    super(message, 401, "UNAUTHORIZED");
    this.name = "AuthError";
  }
}

// Uso
function buscarUsuario(id) {
  const usuario = baseDeDatos.find(u => u.id === id);
  if (!usuario) throw new NotFoundError("Usuario");
  return usuario;
}

try {
  const user = buscarUsuario(999);
} catch (error) {
  if (error instanceof NotFoundError) {
    console.log(`404: ${error.message}`);
  }
}
```

## Manejo de errores en async/await

```javascript
// Opción 1: try/catch (más común)
async function cargarDatos() {
  try {
    const res = await fetch("/api/datos");
    if (!res.ok) throw new AppError("Error HTTP", res.status, "HTTP_ERROR");
    return await res.json();
  } catch (error) {
    console.error("Error:", error.message);
    return null; // valor por defecto
  }
}

// Opción 2: .catch() en la Promise
const datos = await fetch("/api/datos")
  .then(r => r.json())
  .catch(err => {
    console.error(err);
    return [];
  });

// Opción 3: Wrapper reutilizable (patrón Go-style)
async function tryCatch(promise) {
  try {
    const data = await promise;
    return [data, null];
  } catch (error) {
    return [null, error];
  }
}

// Uso del wrapper
const [usuarios, error] = await tryCatch(fetch("/api/users").then(r => r.json()));
if (error) {
  console.error("Falló:", error.message);
} else {
  console.log("Usuarios:", usuarios);
}
```

## Errores silenciosos (anti-patrón ❌)

```javascript
// ❌ NUNCA hagas esto — tragar errores silenciosamente
try {
  operacionCritica();
} catch (e) {
  // vacío = bug invisible
}

// ❌ Tampoco esto
try {
  operacionCritica();
} catch (e) {
  console.log(e); // Console.log no es suficiente en producción
}

// ✅ Mínimo recomendado
try {
  operacionCritica();
} catch (error) {
  console.error("[operacionCritica]", error);
  // En producción: enviar a servicio de monitoreo (Sentry, etc.)
  // reportError(error);
}
```

## Validación defensiva

```javascript
// Validar inputs al inicio de funciones
function crearUsuario({ nombre, email, edad }) {
  // Guard clauses
  if (!nombre?.trim()) throw new ValidationError("nombre", "es requerido");
  if (!email?.includes("@")) throw new ValidationError("email", "formato inválido");
  if (typeof edad !== "number" || edad < 0) throw new ValidationError("edad", "debe ser positivo");

  return {
    nombre: nombre.trim(),
    email: email.toLowerCase(),
    edad,
    creadoEn: new Date(),
  };
}
```

### Ejercicio 1 — Try/catch básico
Escribe una función `parsearJSON(str)` que intente parsear un string como JSON. Si falla, retorna `null` y logea el error.

### Ejercicio 2 — Custom Error
Crea una clase `EdadInvalidaError` que extienda `Error`. Úsala en una función `validarEdad(edad)` que lance el error si la edad es < 0 o > 150.

### Ejercicio 3 — Wrapper tryCatch
Implementa la función `tryCatch` que devuelve `[data, null]` o `[null, error]`. Úsala para hacer un fetch seguro.

### Ejercicio 4 — Validación completa
Crea una función `validarFormulario({ nombre, email, password })` que lance `ValidationError` con mensajes específicos para cada campo inválido. Captura y muestra todos los errores.

#### Ejercicio 2
```javascript
class EdadInvalidaError extends Error {
  constructor(edad) {
    super(`Edad inválida: ${edad}. Debe estar entre 0 y 150.`);
    this.name = "EdadInvalidaError";
    this.edad = edad;
  }
}

function validarEdad(edad) {
  if (typeof edad !== "number" || edad < 0 || edad > 150) {
    throw new EdadInvalidaError(edad);
  }
  return true;
}

try {
  validarEdad(200);
} catch (error) {
  if (error instanceof EdadInvalidaError) {
    console.error(error.message); // "Edad inválida: 200..."
  }
}
```

#### Ejercicio 4
```javascript
class ValidationError extends Error {
  constructor(campo, detalle) {
    super(`${campo}: ${detalle}`);
    this.name = "ValidationError";
    this.campo = campo;
  }
}

function validarFormulario({ nombre, email, password }) {
  const errores = [];

  if (!nombre?.trim()) errores.push(new ValidationError("nombre", "requerido"));
  if (!email?.includes("@")) errores.push(new ValidationError("email", "formato inválido"));
  if (!password || password.length < 8) {
    errores.push(new ValidationError("password", "mínimo 8 caracteres"));
  }

  if (errores.length > 0) {
    const msg = errores.map(e => e.message).join("; ");
    throw new Error(`Validación fallida: ${msg}`);
  }

  return true;
}

try {
  validarFormulario({ nombre: "", email: "bad", password: "123" });
} catch (error) {
  console.error(error.message);
  // "Validación fallida: nombre: requerido; email: formato inválido; password: mínimo 8 caracteres"
}
```
